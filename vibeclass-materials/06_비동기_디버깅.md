2️⃣ Promise 이해하기
Promise란?
비동기 작업의 결과를 나타내는 객체
세 가지 상태:
Pending (대기): 아직 완료 안 됨
Fulfilled (이행): 성공
Rejected (거부): 실패
Promise 생성
const myPromise = new Promise((resolve, reject) => {
  // 비동기 작업 수행
  setTimeout(() => {
    const success = true

    if (success) {
      resolve('성공!')  // 이행
    } else {
      reject(new Error('실패!'))  // 거부
    }
  }, 1000)
})

  results.forEach(result => {
    if (result.status === 'fulfilled') {
      console.log('성공:', result.value)
    } else {
      console.log('실패:', result.reason)
    }
  })
})

}

try-catch 기본
async function fetchData() {
  try {
    const response = await fetch('/api/data')

    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`)
    }

    const data = await response.json()
    return data
  } catch (error) {
    console.error('에러 발생:', error)
    throw error  // 상위로 전파

    console.log('요청 완료')

}

에러 분류 처리
async function apiCall() {
  try {
    const res = await fetch('/api/data')
    const data = await res.json()
    return data
  } catch (error) {
    // 네트워크 에러
    if (error instanceof TypeError) {
      console.error('네트워크 오류')
      return { offline: true }
    }

    // API 에러
    if (error.response) {
      const status = error.response.status
      if (status === 401) {
        console.error('인증 필요')
        // 로그인 페이지로 이동
      } else if (status === 403) {
        console.error('권한 없음')
      } else if (status >= 500) {
        console.error('서버 오류')
      }
    }

    throw error
  }
}

재시도 로직
async function fetchWithRetry(url, options = {}, retries = 3) {
  for (let i = 0; i < retries; i++) {
    try {
      const response = await fetch(url, options)

      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`)
      }

      return await response.json()
    } catch (error) {
      console.log(`시도 ${i + 1}/${retries} 실패:`, error.message)

      if (i === retries - 1) {
        throw error  // 마지막 시도도 실패
      }

      // 지수 백오프 (1초, 2초, 4초...)
      await new Promise(r => setTimeout(r, Math.pow(2, i) * 1000))
    }
  }
}

// 사용

'use client'

import { useState, useEffect } from 'react'

function DataComponent() {
  const [data, setData] = useState(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<Error | null>(null)

  useEffect(() => {
    async function fetchData() {
      try {
        setLoading(true)
        setError(null)

        const res = await fetch('/api/data')
        if (!res.ok) throw new Error('API 오류')

        const json = await res.json()
        setData(json)
      } catch (err) {
        setError(err instanceof Error ? err : new Error('알 수 없는 오류'))
      } finally {
        setLoading(false)
      }
    }

    fetchData()
  }, [])

  if (loading) return <div>로딩 중...</div>
  if (error) return <div>에러: {error.message}</div>
  if (!data) return <div>데이터 없음</div>

  return <div>{JSON.stringify(data)}</div>
}

7️⃣ 로그 전략
로그 레벨
// lib/logger.ts
const LOG_LEVELS = {
  DEBUG: 0,
  INFO: 1,
  WARN: 2,
  ERROR: 3,
} as const

const CURRENT_LEVEL = process.env.NODE_ENV === 'production'
  ? LOG_LEVELS.WARN
  : LOG_LEVELS.DEBUG

export const logger = {
  debug: (...args: any[]) => {
    if (CURRENT_LEVEL <= LOG_LEVELS.DEBUG) {
      console.log('[DEBUG]', new Date().toISOString(), ...args)
    }
  },
  info: (...args: any[]) => {
    if (CURRENT_LEVEL <= LOG_LEVELS.INFO) {
      console.info('[INFO]', new Date().toISOString(), ...args)
    }
  },
  warn: (...args: any[]) => {
    if (CURRENT_LEVEL <= LOG_LEVELS.WARN) {
      console.warn('[WARN]', new Date().toISOString(), ...args)
    }
  },
  error: (...args: any[]) => {
    if (CURRENT_LEVEL <= LOG_LEVELS.ERROR) {
      console.error('[ERROR]', new Date().toISOString(), ...args)
    }
  },
}

// middleware.ts
import { NextRequest, NextResponse } from 'next/server'
import { logger } from './lib/logger'

export function middleware(request: NextRequest) {
  const start = Date.now()

  // 요청 로그
  logger.info(`→ ${request.method} ${request.url}`)

  const response = NextResponse.next()

  // 응답 로그
  const duration = Date.now() - start
  logger.info(`← ${request.method} ${request.url} (${duration}ms)`)

  return response
}

// ✅ 수정 2: null 체크
function FixedComponent2() {
  const [user, setUser] = useState<{ name: string } | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<Error | null>(null)

  useEffect(() => {
    fetch('/api/user')
      .then(res => {
        if (!res.ok) throw new Error('API Error')
        return res.json()
      })
      .then(setUser)
      .catch(setError)
      .finally(() => setLoading(false))
  }, [])

  if (loading) return <div>로딩 중...</div>
  if (error) return <div>에러: {error.message}</div>
  if (!user) return <div>사용자 없음</div>

  return <div>{user.name}</div>  // ✅ 안전
}

// ✅ 수정 3: cleanup 추가
function FixedComponent3() {
  const [data, setData] = useState(null)

  useEffect(() => {
    let isMounted = true  // ✅ 마운트 상태 추적

    const timer = setInterval(async () => {
      try {
        const res = await fetch('/api/data')
        const json = await res.json()
        if (isMounted) {  // ✅ 마운트된 경우만 업데이트
          setData(json)
        }
      } catch (error) {
        console.error('Fetch error:', error)
      }
    }, 1000)

    return () => {  // ✅ cleanup
      isMounted = false
      clearInterval(timer)
    }
  }, [])

  return <div>{data}</div>
}
