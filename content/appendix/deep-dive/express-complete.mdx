## 1. Express 시작하기

### 프로젝트 세팅

```bash
mkdir my-express-app
cd my-express-app

npm init -y
npm install express cors dotenv morgan
npm install -D nodemon
```

```json
{
  "name": "my-express-app",
  "type": "module",
  "scripts": {
    "start": "node index.js",
    "dev": "nodemon index.js"
  }
}
```

### 첫 번째 서버

```javascript
// index.js
import express from 'express'
import cors from 'cors'
import morgan from 'morgan'

const app = express()
const PORT = process.env.PORT || 3000

// 미들웨어 설정
app.use(cors())
app.use(morgan('dev'))
app.use(express.json())

// 라우트
app.get('/', (req, res) => {
  res.json({ message: 'Hello Express!' })
})

app.listen(PORT, () => {
  console.log(`서버 실행: http://localhost:${PORT}`)
})
```

## 2. 라우팅

```javascript
// 기본 라우팅
app.get('/users', (req, res) => { ... })
app.post('/users', (req, res) => { ... })
app.put('/users/:id', (req, res) => { ... })
app.patch('/users/:id', (req, res) => { ... })
app.delete('/users/:id', (req, res) => { ... })

// URL 파라미터
app.get('/users/:id', (req, res) => {
  const { id } = req.params  // URL의 :id 값
  res.json({ userId: id })
})

// 쿼리 파라미터
app.get('/users', (req, res) => {
  const { page = 1, limit = 10, search } = req.query
  // /users?page=2&limit=20&search=홍길동
  res.json({ page, limit, search })
})
```

### Router로 모듈화

```javascript
// routes/users.js
import { Router } from 'express'
const router = Router()

router.get('/', (req, res) => {
  res.json({ users: [] })
})

router.post('/', (req, res) => {
  const { name, email } = req.body
  res.status(201).json({ id: 1, name, email })
})

export default router

// index.js
import usersRouter from './routes/users.js'
app.use('/api/users', usersRouter)
```

## 3. 미들웨어 이해하기

미들웨어는 요청(Request)과 응답(Response) 사이에서 실행되는 함수입니다.

```javascript
// 미들웨어 패턴: (req, res, next)
const logger = (req, res, next) => {
  console.log(`[${new Date().toISOString()}] ${req.method} ${req.url}`)
  next()  // 다음 미들웨어로 이동
}

app.use(logger)  // 전역 미들웨어

// 특정 경로에만 적용
app.use('/api', authMiddleware)

// 인증 미들웨어 예시
const authMiddleware = (req, res, next) => {
  const token = req.headers.authorization?.split(' ')[1]

  if (!token) {
    return res.status(401).json({ error: '인증이 필요합니다' })
  }

  try {
    const decoded = verifyToken(token)
    req.user = decoded
    next()
  } catch {
    res.status(401).json({ error: '유효하지 않은 토큰입니다' })
  }
}
```

## 6. 에러 처리

Express에서 에러 처리 미들웨어는 인자가 4개입니다.

```javascript
// 에러 처리 미들웨어 (반드시 마지막에 등록)
app.use((err, req, res, next) => {
  console.error(err.stack)

  const statusCode = err.statusCode || 500

  res.status(statusCode).json({
    error: {
      code: err.code || 'INTERNAL_ERROR',
      message: err.message || '서버 오류가 발생했습니다'
    }
  })
})

// 비동기 에러 처리 (express 5.x는 자동, 4.x는 수동)
const asyncHandler = (fn) => (req, res, next) => {
  Promise.resolve(fn(req, res, next)).catch(next)
}

app.get('/users', asyncHandler(async (req, res) => {
  const users = await User.findAll()  // 여기서 에러 발생 시 자동으로 에러 미들웨어로
  res.json(users)
}))
```

## 7. 실전 CRUD API 예제

```javascript
// 메모리 기반 간단한 CRUD
let users = [
  { id: 1, name: '김철수', email: 'kim@example.com' },
  { id: 2, name: '이영희', email: 'lee@example.com' }
]
let nextId = 3

// GET /api/users
router.get('/', (req, res) => {
  res.json({ data: users, total: users.length })
})

// GET /api/users/:id
router.get('/:id', (req, res) => {
  const user = users.find(u => u.id === Number(req.params.id))
  if (!user) return res.status(404).json({ error: '사용자를 찾을 수 없습니다' })
  res.json(user)
})

// POST /api/users
router.post('/', (req, res) => {
  const { name, email } = req.body
  if (!name || !email) {
    return res.status(400).json({ error: 'name과 email이 필요합니다' })
  }
  const newUser = { id: nextId++, name, email }
  users.push(newUser)
  res.status(201).json(newUser)
})

// PATCH /api/users/:id
router.patch('/:id', (req, res) => {
  const index = users.findIndex(u => u.id === Number(req.params.id))
  if (index === -1) return res.status(404).json({ error: '사용자를 찾을 수 없습니다' })
  users[index] = { ...users[index], ...req.body }
  res.json(users[index])
})

// DELETE /api/users/:id
router.delete('/:id', (req, res) => {
  const index = users.findIndex(u => u.id === Number(req.params.id))
  if (index === -1) return res.status(404).json({ error: '사용자를 찾을 수 없습니다' })
  users.splice(index, 1)
  res.status(204).end()
})
```

## 핵심 포인트 정리

- Express의 미들웨어는 `(req, res, next)` 패턴입니다. `next()`를 호출해야 다음 미들웨어로 넘어갑니다
- 에러 처리 미들웨어는 인자가 4개 `(err, req, res, next)`이고 반드시 마지막에 등록해야 합니다
- Router를 사용해 기능별로 라우트를 분리하면 코드 유지보수가 쉬워집니다
- 응답 상태 코드는 의미 있게 사용하세요. 201(생성), 400(잘못된 요청), 404(없음), 409(충돌)
